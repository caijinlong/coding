void swap (int i, int j, int * p) {
    if (i == j) return;
    int foo = p[i];
    p[i] = p[j];
    p[j] = foo;
}

class Solution {
public:
    int removeDuplicates(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (n == 0) return 0;
        
        int pre_value = A[0];
        int valid_index = 0;
        int len = 1;
        
        for (int i = 1; i < n; i++) {
            if (A[i] == pre_value) continue;
            else {
              pre_value = A[i];
              len ++;
              swap(valid_index + 1, i, A);
              valid_index++;
            }
        }
        return len;
    }
};
